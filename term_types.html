Term types

lightQL is fully compatible with Elasticsearch in regards to the kind of terms supported. See below some examples of these terms:

scala> import com.tecsisa.lightql.parser._
import com.tecsisa.lightql.parser._

scala> parse("foo = 25") // integer
res0: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,=,25)),8)

scala> parse("foo = 2.0000000000000002") // double
res1: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,=,2.0)),24)

scala> parse("foo = \"foobar\"") // string
res2: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,=,foobar)),14)

scala> parse("foo = [25, -2.4, \"foobar\"]") // heterogeneous list
res3: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,=,List(25, -2.4, foobar))),26)

scala> parse("foo = 2001-07-12") // date
res4: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,=,2001-07-12T00:00:00.000Z)),16)

scala> parse("foo = 2001-07-12T12:10:30.002+02:00") // date-time with timezone
res5: com.tecsisa.lightql.parser.StringParsed[com.tecsisa.lightql.ast.Query] = Success(Query(Clause(foo,=,2001-07-12T10:10:30.002Z)),35)
